

fun main() {
fun fib(a) {
   if a <= 1 { return 1 }

   return fib(a-1) + fib(a-2);
}


let startTime = _time;
print 'fibonacci(4) = ' + fib(4)
print 'took: ' + (_time - startTime)*1000 + 'ms'
print '' 

startTime = _time;
print 'fibonacci(20) = ' + fib(20)
print 'took: ' + (_time - startTime)*1000 + 'ms'
print  ''

startTime = _time;
print 'fibonacci(30) = ' + fib(30)
print 'took: ' + (_time - startTime)*1000 + 'ms'
}

main()
